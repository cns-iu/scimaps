import { Component, HostBinding, Input } from '@angular/core';
import { ActionCardItem } from '../../../core/models/action-card-item';

@Component({
  selector: 'sci-action-card',
  templateUrl: './action-card.component.html',
  styleUrls: ['./action-card.component.scss']
})
export class ActionCardComponent {
  /** HTML class name */
  @HostBinding('class') readonly clsName = 'sci-action-card';

  @Input() actionItem!: ActionCardItem;
  closed = false;

  get imageUrl(): string {
    return `url('${this.actionItem.imageSource}')`;
  }

  goToLink(link: string): void {
    window.open(link, '_blank');
  }
}

<div class="action-card" [class.hidden]="closed">

    <div class="body">{{ actionItem.body }}</div>

    <div class="action-button" (click)="goToLink(actionItem.buttonLink)">{{ actionItem.buttonLabel }}</div>
  </div>

  <div class="image-container" [style.background-image]="imageUrl">
  </div>

  <mat-icon class="close-icon" (click)="closed = true">clear</mat-icon>
</div>
